536 public static object RankingCheck(string value, string type)
537 {
538 var objectJson = DynamicJson.Parse(value);
539 object ret = null;
540 if (objectJson.IsDefined("Code"))
541 {
542 string error = "Eoor Code:" + objectJson["Code"] + " " + objectJson["Message"];
543
544 throw new APIResponsesException(error);
545 }
546 switch (type)
547 {
548 case TYPE.PriceIncreaseRate:
549 case TYPE.PriceDropRate:
550 case TYPE.TopTradingVolume:
551 case TYPE.TradingPrice:
552 ret = RankingDefaultToArray(objectJson);
553 break;
554 case TYPE.TickCount:
555 ret = RankingBytickCountToArray(objectJson);
556 break;
557 case TYPE.RapidIncreaseInTradingVolume:
558 ret = RankingByTradeVolumnToArray(objectJson);
559 break;
560 case TYPE.RapidIncreaseInTradingValue:
561 ret = RankingByTradeValueToArray(objectJson);
562 break;
563 case TYPE.IncreasedCreditSales:
564 case TYPE.CreditSalesLoss:
565 case TYPE.IncreasedCreditPurchase:
566 case TYPE.CreditPurchaseBalanceReduction:
567 case TYPE.CreditHighMagnification:
568 case TYPE.CreditLowMagnification:
569 ret = RankingByMarginToArray(objectJson);
570 break;
571 case TYPE.PriceIncreaseRateByCategory:
572 case TYPE.PriceReductionRateByCategory:
573 ret = RankingByCategoryToArray(objectJson);
574 break;
575 default:
576 ret = RankingDefaultToArray(objectJson);
577 break;
578 }
579
580 return ret;
581 }
