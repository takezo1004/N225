522 string side;

648 private void SendOrderAuto(InputOrder fileld)
649 {
650 string ms = "売買シグナルを受信しました: " + fileld.Strategy + " " + Convert.ToString(fileld.Interval) + " " +
651 new CashMargin(fileld.TradeType + 1).DisplayValue + " " +
652 new Domain.ValueObjects.Side(fileld.Side).DisplayValue + " " + Convert.ToString(fileld.Price);
653 WriteMessage(ms);
654
655 //form Strategy gridviewにトレードサインを表示する
656 StrategyManeger.UpDate(StrategyViews,fileld);
657
658 if (AutoButoon == false)
659 {
660 return;
661 }
662 if (StrategyManeger.IsTrade(fileld.Strategy, fileld.Interval) == false)
663 {
664 WriteMessage("Strategyはチェックなし又は登録されていません。");
665 return;
666 }
667 //返済の場合ExecutionIdを取得する
668 List<string> listid;
669 string executionId = String.Empty;
670 if (fileld.TradeType == TradeType.ExitOrder)
671 {
672 listid = PositionManager.GetExecutionId(fileld.Strategy,
673 fileld.Interval, fileld.Side);
674
675 if (listid.Count == 0)
676 {
677 WriteMessage("返済するポジションはありません。");
678 return;
679 }
680 foreach (string id in listid)
681 {
682 fileld.ExecutionId = id;
683 SendOrder(fileld);
684 }
685 }
686 else
687 {
688 SendOrder(fileld);
689 }

697 private void ProfitCalculate(double currentPrice)
698 {
699 if (currentPrice == 0)
700 {
701 return;
702 }
703 int rowCount = PositionList.Count;
704 double qty;
705 string side;
706 double price;
707 double profitPrice;
708 for (int i = 0; i < rowCount; i++)
709 {
710 var entity = PositionList.ElementAt(i);
711 qty = entity.LeaveQty;
712 price = Convert.ToDouble(entity.Price);
713 side = entity.Side;
714
715 if (side == "買")
716 {
717 profitPrice = (currentPrice - price) * qty * 100;
718 }
719 else
720 {
721 profitPrice = (price - currentPrice) * qty * 100;
722 }
723 PositionList.ElementAt(i).Profit = profitPrice;
724 }
725 }

768 private void InquiryOrderEventHandler(object sender, OrdersResultModel e)
769 {
770 //double型のプロパティはobject型に変更した。price.qty　他
771 //型変換して使う Comnver.toInt32(price)
772 string _executionID;
773 OrdersResultModel _ordersResul = e;
774 string _orderID = e.ID;
775
776 int count = e.Details.Count;
777 double _price = Convert.ToDouble(e.Price);
778 int _orderState = e.OrderState;
779 int _reciveType = e.Details[count - 1].RecType;
780 int _cashMargin = e.CashMargin;
781 if (_cashMargin == 2)
782 {
783 _executionID = e.Details[count - 1].ExecutionID;
784 if (_reciveType == 8)
785 {
786 _price = Convert.ToDouble(e.Details[count - 1].Price);
787 }
788 }
789 else
790 {
791 _executionID = OrderManager.GetExecutionId(_orderID);
792 if (_reciveType == 8)
793 {
794 _price = Convert.ToDouble(e.Details[count - 1].Price);
795 }
796 }
797
798 //updataデータ作成
799 OrederManagerEntity _orderEntity = new OrederManagerEntity()
800 {
801 Id = _orderID,
802 ReciveType = _reciveType,
803 OrderQty = Convert.ToDouble(e.OrderQty),
804 CumQty = Convert.ToDouble(e.CumQty),
805 Price = _price
806 };
807
808 PositionManagerEntity _positionEntity = new PositionManagerEntity()
809 {
810 OrderID = _orderID,
811 ExecutionID = _executionID,
812 OrderState = _orderState,
813 ReciveType = _reciveType,
814 CashMargin = _cashMargin,
815 OrderQty = Convert.ToDouble(e.OrderQty),
816 CumQty = Convert.ToDouble(e.CumQty),
817 Price = _price
818 };
819
820 if (_reciveType == 3)
821 {
822 OrderManager.Xepired(_orderEntity);
823 PositionManager.Xepired(_positionEntity);
824 OrderMessage(_positionEntity);
825 }
826 if (_reciveType == 4)
827 {
828 OrderManager.Order(_orderEntity);
829 }
830 if (_reciveType == 6 || _reciveType == 7)
831 {
832 OrderManager.Cancel(_orderEntity);
833 PositionManager.Cancel(_positionEntity);
834 OrderMessage(_positionEntity);
835 }
836 if (_reciveType == 8 && _orderState == 3)
837 {
838 OrderManager.Contrct(_orderEntity);
839 PositionManager.Contrct(_ordersResul, _positionEntity);
840 OrderMessage(_positionEntity);
841 }
842
843
844 if (_reciveType == 8 && _orderState == 5)
845 {
846 OrderManager.ContrctAll(_orderEntity);
847 PositionManager.ContrctAll(_positionEntity);
848 OrderMessage(_positionEntity);
849 }
