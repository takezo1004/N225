294 public static void Contrct(OrdersResultModel resultModel, PositionManagerEntity data)
295 {
296 string _orderID = resultModel.ID;
297 OrderListEntity _entity = OrdersCache.GetsEntity(_orderID);
298 int tradeMode = _entity.TradeMode.Value;
299
300 if (resultModel.CashMargin == 2)
301 {
302 foreach (OrderDetails detail in resultModel.Details)
303 {
304 if (detail.RecType == 8)
305 {
306 PositionListEntity viewEntity = new PositionListEntity(
307 tradeMode,
308 Convert.ToInt32(_entity.RecvTime.yyyyMMdd), //Convert.ToInt32(_entity.RecvTime.Value),
309 _entity.Strategy,
310 _entity.Interval,
311 _entity.Side.Value,
312 Convert.ToDouble(detail.Qty),
313 Convert.ToDouble(0),
314 Convert.ToDouble(detail.Price),
315 0,
316 detail.ExchangeID,
317 _orderID);
318
319 Add(viewEntity);
320 }
321
322 }
323 }
324 else
325 {
326 int count = resultModel.Details.Count;
327 string _executionID = resultModel.Details[count - 1].ExecutionID;
328 double qty = Convert.ToDouble(resultModel.Details[count - 1].Qty);
329 int row = GetPositionListIndex(_executionID);
330
331 double leaveQty = PositionList.ElementAt(row).LeaveQty;
332 double holdQty = PositionList.ElementAt(row).HoldQty;
333
334
335 if (leaveQty > holdQty)
336 {
337 Dispatcher.Invoke(delegate ()
338 {
339 double _leaveQty = PositionList.ElementAt(row).LeaveQty;
340 PositionList.ElementAt(row).LeaveQty -= qty;
341 PositionList.ElementAt(row).HoldQty -= qty;
342
343 //cash更新 PositionManagerEntityを使う CumQty -> LeaveQty,OrderQty ->HoldQtyとする
344 data.CumQty -= qty;
345 data.OrderQty -= qty;
346 PositionsCache.UpDate(data);
347 });
348 }
349 if (PositionList.ElementAt(row).LeaveQty <= 0)
350 {
351 Deleted(data, tradeMode);
352 }
353 }
354 }
