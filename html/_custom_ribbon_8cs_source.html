94 public void OnLoad(IRibbonUI ribbon)
95 {
96
97 XElement xml = XElement.Load(xmlPath);
98 IEnumerable<XElement> infos = from item in xml.Elements("setting1")
99 select item;
100
101 foreach (XElement info in infos)
102 {
103 _orderPressed = info.Element(RibbonItem.発注制御ボタン).Value == "1" ? true : false;
104 _updatePressed = info.Element(RibbonItem.更新制御ボタン).Value == "1" ? true : false;
105 _port = info.Element(RibbonItem.ポート入力ボックス).Value;
106 }
107
108 _thisRibbon = ribbon ?? throw new ArgumentNullException(nameof(ribbon));
109
110 settingUpdate("ApiToken", "");
111
112 }

304 private string getToken (string apiPassword, ref bool token)
305 {
306 var strJson = ExcelFunctionController.KABUSUTE_API_TOKEN(apiPassword);
307 string response = "";
308
309 try
310 {
311 var objectJson = DynamicJson.Parse(strJson);
312
313 if (objectJson.IsDefined("ResultCode"))
314 {
315 if (objectJson.ResultCode == 0)
316 {
317 token = true;
318 return objectJson.Token;
319 }
320 else
321 {
322 token = false;
323 MessageBox.Show("不明なエラーです");
324 }
325 }
326 else if(objectJson.IsDefined("Code"))
327 {
328 token = false;
329 MessageBox.Show("Code " + objectJson.Code +":" + objectJson.Message);
330 }
331 }
332 catch (Exception exception)
333 {
334 if(exception.InnerException == null)
335 MessageBox.Show(strJson);
336 else
337 MessageBox.Show(exception.InnerException.ToString());
338
339 }
340
341 return response;
342
343 }
